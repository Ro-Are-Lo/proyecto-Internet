// VIEWMODEL
// Aquí manejamos el "estado" y la "lógica de interacción".
// Los ViewModels usan hooks para preparar datos y exponerlos a las Views.

import { useEffect, useState } from "react";
import { User } from "../models/User";
import { fetchUsers } from "../services/userService";

export function useUserViewModel() {
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);

  // Al montar el componente, pedimos los usuarios
  useEffect(() => {
    fetchUsers().then(data => {
      setUsers(data);
      setLoading(false);
    });
  }, []);

  // Devolvemos solo lo que la View necesita
  return { users, loading };
}
